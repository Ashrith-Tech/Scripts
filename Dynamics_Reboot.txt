# ----------------- FUNCTIONS -----------------

function Log {
    param([string]$msg)
    $txtOutput.AppendText("$(Get-Date -Format 'HH:mm:ss') - $msg`r`n")
    $txtOutput.Refresh()
}

function Wait-ForServerOnline {
    param(
        [string]$ComputerName,
        [int]$PingTimeout = 900,     # 15 min
        [int]$WinRMTimeout = 600     # 10 min
    )

    Log "Waiting for $ComputerName to respond to ping..."
    $stopwatch = [System.Diagnostics.Stopwatch]::StartNew()

    # ----------------------------
    # Wait for Ping
    # ----------------------------
    while ($stopwatch.Elapsed.TotalSeconds -lt $PingTimeout) {
        if (Test-Connection -ComputerName $ComputerName -Count 1 -Quiet) {
            Log "$ComputerName ping response detected."
            break
        }
        Start-Sleep -Seconds 5
    }

    if ($stopwatch.Elapsed.TotalSeconds -ge $PingTimeout) {
        Log "[ERROR] $ComputerName did NOT respond to ping in $PingTimeout seconds."
        return $false
    }

    # ----------------------------
    # Wait for WinRM
    # ----------------------------
    Log "Waiting for WinRM to become available on $ComputerName..."
    $winRMStart = Get-Date

    while ((New-TimeSpan -Start $winRMStart -End (Get-Date)).TotalSeconds -lt $WinRMTimeout) {

        try {
            Test-WSMan -ComputerName $ComputerName -ErrorAction Stop | Out-Null
            Log "WinRM is online for $ComputerName."
            return $true
        }
        catch {
            Start-Sleep -Seconds 5
        }
    }

    Log "[WARNING] WinRM did NOT become available within $WinRMTimeout seconds. Continuing anyway."
    return $false
}

function Validate-AutomaticServices {
    param($Servers)

    $results = @()

    foreach ($server in $Servers) {
        Log "Validating Automatic services on $server..."

        try {
            $services = Get-Service -ComputerName $server |
                        Where-Object { $_.StartType -eq "Automatic" }

            foreach ($svc in $services) {
                if ($svc.Status -ne "Running") {
                    $results += [PSCustomObject]@{
                        Server      = $server
                        ServiceName = $svc.Name
                        DisplayName = $svc.DisplayName
                        Status      = $svc.Status
                    }
                }
            }
        }
        catch {
            Log "Error checking services on $server : $_"
        }
    }

    return $results
}


# ----------------- MAIN PROCESS -----------------

$btnStart.Add_Click({

    # Read server names properly
    $Servers = $txtServers.Lines |
               ForEach-Object { $_.Trim() } |
               Where-Object { $_ -ne "" }

    if ($Servers.Count -eq 0) {
        [System.Windows.Forms.MessageBox]::Show("Please enter at least one server.")
        return
    }

    Log "Starting reboot sequence for $($Servers.Count) servers..."

    foreach ($server in $Servers) {

        Log "Sending reboot command to $server ..."
       try {
    Invoke-Command -ComputerName $server -ScriptBlock {
        Restart-Computer -Force
    } -ErrorAction Stop

    Log "$server reboot command sent successfully."
}
catch {
    Log "[ERROR] Could not reboot $server : $($_.Exception.Message)"
    continue
}


        # Wait until server is online
        Wait-ForServerOnline -ComputerName $server | Out-Null

        Log "Waiting 5 minutes before rebooting next server..."
        Start-Sleep -Seconds 300   # <<< 5 minutes
    }

    Log "All reboots completed. Starting Automatic services validation..."

    if (!(Test-Path "C:\Temp")) { New-Item -ItemType Directory -Path "C:\Temp" | Out-Null }

    $csvPath = "C:\Temp\AutomaticServiceValidation.csv"
    $results = Validate-AutomaticServices -Servers $Servers
    $results | Export-Csv -Path $csvPath -NoTypeInformation

    Log "Validation complete. CSV saved: $csvPath"

    [System.Windows.Forms.MessageBox]::Show("Process Completed! Report saved to $csvPath")
})
